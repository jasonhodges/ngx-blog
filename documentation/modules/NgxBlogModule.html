  { path: '', component: HomeComponent }

  imports: [
    CommonModule,
    PostModule.forRoot(),
    MarkdownToHtmlModule.forRoot()

  exports: [],
  declarations: [
    HomeComponent,
    PostsComponent,
    BlogComponent
  ],
  providers: [
    PostsService
  ],
})
